* Config
#+BEGIN_SRC emacs-lisp :noweb tangle
(message "[krista] Loading magit...")
(use-package magit
  :config
  <<magit-color-scheme>>
  ;; Default arguments to magit log
  (setq magit-log-arguments '("-n150" "--graph" "--decorate" "--color"))
  :bind
  (("C-c g" . magit-status)))
#+END_SRC
